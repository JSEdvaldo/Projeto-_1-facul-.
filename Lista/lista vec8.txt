#include <iostream>

using namespace std;

void ler_vetor(int v1[],int tam){
    for(int i = 0 ; i< tam; i ++){
        cin >> v1[i];       // digitar os valores do vetor.
    }
}
bool verificar(int v1[], int v2[],int tam){
    for(int i = 0; i < tam; i++){
        if(v1[i] != v2[i]){
            return false;
        }
    }
    return true;
}

int main(){

const int tam = 3; //pode ser qual quer tamanho.

int vetor1[tam];
int vetor2[tam];

ler_vetor(vetor1,tam);
ler_vetor(vetor2,tam);

cout << verificar(vetor1,vetor2,tam);

}
